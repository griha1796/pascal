program massiv;
var
a:array [1..100] of integer; n, i, j, p:integer;
begin
write('n=');
readln (n);
writeln('Исходный массив');
for i:=1 to n do
begin
a[i]:= random (10);
write (a[i],'');
end;
writeln();
writeln('Получившийся массив');
for i:=1 to n-1 do
for j:=i+1 to n do
if a[j] p:=a[i];
a[i]:=a[j];
a[j]:=p;
end;
for i:=1 to n do
write (a[i],'');
end.
